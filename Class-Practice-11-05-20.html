set.seed(1234)
dplyr::sample_n(blood_pressure,27)
levels(blood_pressure$drug)
## NULL
blood_pressure$drug <- ordered(blood_pressure$drug, levels = c("A", "B", "C"))
library(dplyr)
group_by(blood_pressure, drug) %>%
  summarise(
    count = n(),
    mean = mean(bloodPressure, na.rm = TRUE),
    sd = sd(bloodPressure, na.rm = TRUE)
  )
## `summarise()` ungrouping output (override with `.groups` argument)
ggboxplot(blood_pressure, x = "drug", y = "bloodPressure", color = "drug", palette = c("#bb5a00", "#bb9f00", "#00bb22"), order = c("A", "B", "C"), ylab= "Drug", xlab = "Blood Pressure", main="Blood Pressure Box plot Data Set")

ggline(blood_pressure, x="drug", y= "bloodPressure", add = c("mean_se", "jitter"), order = c("A", "B", "C"), ylab="Drug", xlab = "Blood Pressure", main="Mean Plot with 95% CI")

#One-way ANOVA Test
bp.aov <- aov(bloodPressure ~ drug, data = blood_pressure)

summary(bp.aov)
